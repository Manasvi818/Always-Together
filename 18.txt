<script>
document.addEventListener("DOMContentLoaded", () => {
    loadMessages();
});

/* SAVE */
function saveMessages() {
    const msgs = [];
    document.querySelectorAll(".message-text").forEach(m => msgs.push(m.textContent));
    localStorage.setItem("messages", JSON.stringify(msgs));
}

/* LOAD */
function loadMessages() {
    const saved = JSON.parse(localStorage.getItem("messages")) || [];
    saved.forEach(msg => addMessage(msg, false));
}

/* SEND */
function sendMessage() {
    const input = document.getElementById("messageInput");
    if (!input.value.trim()) return;

    addMessage(input.value.trim(), true);
    heartBurst();
    saveMessages();
    input.value = "";
}

/* ADD MESSAGE */
function addMessage(text, animate=true) {
    const list = document.getElementById("messageList");

    const card = document.createElement("div");
    card.className = "message-card";
    if (!animate) card.style.animation = "none";

    card.innerHTML = `
        <span class="message-text">${text}</span>
        <div class="action-buttons">
            <span onclick="editMessage(this)">‚úèÔ∏è</span>
            <span onclick="deleteMessage(this)">üóëÔ∏è</span>
        </div>
    `;

    list.appendChild(card);
    list.scrollTop = list.scrollHeight;
}

/* DELETE */
function deleteMessage(btn) {
    btn.closest(".message-card").remove();
    saveMessages();
}

/* EDIT */
function editMessage(btn) {
    const span = btn.closest(".message-card").querySelector(".message-text");
    const updated = prompt("Edit Message:", span.textContent);
    if (updated && updated.trim()) {
        span.textContent = updated.trim();
        saveMessages();
    }
}

/* HEART BURST */
function heartBurst() {
    const btn = document.querySelector(".send-btn");
    const rect = btn.getBoundingClientRect();

    for (let i = 0; i < 6; i++) {
        const heart = document.createElement("div");
        heart.className = "burst-heart";
        heart.textContent = "‚ù§";

        heart.style.left = `${rect.left + rect.width/2}px`;
        heart.style.top = `${rect.top}px`;

        heart.style.setProperty("--driftX",(Math.random()-0.5)*140+"px");
        heart.style.setProperty("--duration",(0.8+Math.random()*0.6)+"s");

        document.body.appendChild(heart);
        setTimeout(()=>heart.remove(),1200);
    }
}
